2. IndexController

    ```Java
     @Controller
     public class IndexController {
         //让 / 和 login 请求都跳转到 login请求
         @GetMapping(value = {"/","login"})
         public String loginPage(){
             return "login";
         }
         @PostMapping("/login")
         public String main(User user, HttpSession session, Model model){
             //⭐对 用户名 和 密码 做判断
             if(StringUtils.hasLength(user.getUserName()) && "123456".equals(user.getPassword())){
                 //把登陆成功的用户保存起来
                 session.setAttribute("loginUser",user);
                 //登陆成功：重定向到main.html页面，防止表单重复提交
                 return "redirect:main.html";
             }else {
                 //登陆失败回到登陆页面
                 model.addAttribute("msg","密码错误，请重新登录");
                 return "login";
             }
         }
         @GetMapping("/main.html")
         public String mainPage(HttpSession session,Model model){
             Object loginUser = session.getAttribute("loginUser");
             //⭐判断是否登录
             if (loginUser != null){
                 return "main";
             }else {
                 //如果当前没有登录：返回登录页
                 //同时有一个前端提示
                 model.addAttribute("msg","未登录，请先登录");
                 return "login";
             }
         }
     }
    ```

4. login.html

    ```HTML
     <!--要使用thymeleaf 要引入名称空间-->
     <html lang="en" xmlns:th="http://www.thymeleaf.org">
     <!--⭐表单提交的 action（请求url）改为 /login 与当前（根）路径的拼接-->
     <form class="form-signin" action="main.html" method="post" th:action="@{/login}">
         ...
         <div class="login-wrap">
             <!--⭐label 用于提示信息，将存在域中的 提示信息 msg 提出来显示-->
             <label style="color: red" th:text="${msg}"></label>
             <!--⭐给提交的 input 添加 name，底层会自动封装到 main() 中的 user 对象中-->
             <input type="text" name="userName" class="form-control" placeholder="User ID" autofocus>
             <input type="password" name="password" class="form-control" placeholder="Password">
             <button class="btn btn-lg btn-login btn-block" type="submit">
                 <i class="fa fa-check"></i>
             </button>
    ```
