1. 步骤：
    
    1. 修改RequestParams类，添加brand、city、starName、minPrice、maxPrice等参数
        
        ```Java
         @Data
         public class RequestParams {
             private String key;
             private Integer page;
             private Integer size;
             private String sortBy;
             private String brand;
             private String startName;
             private String city;
             private String minPrice;
             private String maxPrice;
         }
        ```
        
    
    1. 修改search方法的实现，在关键字搜索时，如果brand等参数存在，对其做过滤
        
        1. 过滤条件包括：
            
            1. city精确匹配
            
            1. brand精确匹配
            
            1. starName精确匹配
            
            1. price范围过滤
            
        
        1. 注意事项：
            
            1. ==多个条件之间是AND关系，组合多条件用====**BooleanQuery**==
            
            1. 参数存在才需要过滤，做好非空判断
            
        
        1. 代码：
            
            ```Java
             @Service
             public class HotelService extends ServiceImpl<HotelMapper, Hotel> implements IHotelService {
                 @Autowired
                 private RestHighLevelClient client;
             
                 @Override
                 public PageResult search(RequestParams params) {
                     try {
                         //1. 准备 request
                         SearchRequest request = new SearchRequest("hotel");
                         //2. ⭐准备 DSL
                         //2.1 query（条件过滤需要使用复合查询 boolQuery）
                         //构建 boolQuery
                         BoolQueryBuilder boolQuery = QueryBuilders.boolQuery();
             
                         //关键字搜索（must）
                         String key = params.getKey();
                         if (key == null || key.equals("")) {
                             boolQuery.must(QueryBuilders.matchAllQuery());
                         }else {
                             boolQuery.must(QueryBuilders.matchQuery("all", key));
                         }
             
                         //城市条件
                         String city = params.getCity();
                         if (city != null && !city.equals("")) {
                             boolQuery.filter(QueryBuilders.termQuery("city", city));
                         }
             
                         //品牌条件
                         String brand = params.getBrand();
                         if (brand != null && !brand.equals("")) {
                             boolQuery.filter(QueryBuilders.termQuery("brand", brand));
                         }
             
                         //星级条件
                         String startName = params.getStartName();
                         if (startName != null && !startName.equals("")) {
                             boolQuery.filter(QueryBuilders.termQuery("startName", startName));
                         }
             
                         //价格条件
                         String maxPrice = params.getMaxPrice();
                         String minPrice = params.getMinPrice();
                         if (maxPrice != null && !maxPrice.equals("") && minPrice != null && !minPrice.equals("")) {
                             boolQuery.filter(QueryBuilders.rangeQuery("price").lte(maxPrice).gte(minPrice));
                         }
             
                         request.source().query(boolQuery);
             
                         //2.2 分页
                         int page = params.getPage();
                         int size = params.getSize();
                         request.source().from((page - 1) * size).size(size);
             
                         //3. 发送请求
                         SearchResponse response = client.search(request, RequestOptions.DEFAULT);
             
                         //4. 解析响应
                         return handleResponse(response);
                     } catch (IOException e) {
                        throw new RuntimeException(e);
                     }
                 }
             
                 private PageResult handleResponse(SearchResponse response) {
                     //4.1 获取总结果数
                     SearchHits searchHits = response.getHits();
                     long total = searchHits.getTotalHits().value;
             
                     //4.2 获取数组文档
                     SearchHit[] hits = searchHits.getHits();
                     // 遍历
                     ArrayList<HotelDoc> hotelDocs = new ArrayList<>();
                     for (SearchHit hit : hits) {
                         // 获取 json 文档
                         String source = hit.getSourceAsString();
                         // 反序列化
                         HotelDoc hotelDoc = JSON.parseObject(source, HotelDoc.class);
                         hotelDocs.add(hotelDoc);
                     }
                     return new PageResult(total,hotelDocs);
                 }
             }
            ```
