     @Override
     public Result login(LoginFormDTO loginForm, HttpSession session) {
         //1. 校验手机号
         String phone = loginForm.getPhone();
         if (RegexUtils.isPhoneInvalid(phone)){
             //2. 如果不符合，返回错误信息
             return Result.fail("手机号格式错误！");
         }
         //2. 校验验证码
         Object cacheCode = session.getAttribute("code");
         String code = loginForm.getCode();
         if (cacheCode == null || !cacheCode.toString().equals(code)){
             //3. 不一致报错
             return Result.fail("验证码错误");
         }
         //4. 一致，根据手机号查询用户
         User user = query().eq("phone", phone).one();   # mybatisplus 写法
         //5. 判断用户是否存在
         if (user == null){
             //6. 不存在，创建新用户并保存
             user = createUserWritePhone(phone);
         }
     
         //7. 保存用户信息到session中
         //⭐只向前端返回少量的信息，避免泄露隐私密码
         //使用hutool工具类，将User对象对应数据拷贝到UserDTO对象中
         UserDTO userDTO = BeanUtil.copyProperties(user, UserDTO.class);
         session.setAttribute("user", userDTO);
     
         //⭐这里并不需要返回登录凭证，使用的是session实现的登录，
         // 只要再次访问时可以使用cookie中的sessionID能够得到session即登陆成功
         return Result.ok();
     }
     
     private User createUserWritePhone(String phone) {
         //创建用户
         User user = new User();
         user.setPhone(phone);
         user.setNickName("user_" + RandomUtil.randomString(10));
         //保存用户
         save(user);
         return user;
     }
