# 一、代码实现
- 示例：基于Redis的Stream结构作为消息队列，实现异步秒杀下单
    1. 创建一个Stream类型的消息队列，名为stream.orders
    2. 修改之前的秒杀下单Lua脚本，在认定有抢购资格后，直接向stream.orders中添加消息，内容包含voucherId、userId、orderId
    3. 项目启动时，开启一个线程任务，尝试获取stream.orders中的消息，完成下单
1. 首先创建一个队列和组

    ```Bash
     127.0.0.1:6379> XGROUP CREATE stream.orders g1 0 MKSTREAM
     OK
    ```
2. 由于现在使用的是 redis 的stream消息队列，故生产者向队列中发送消息也是对redis的操作，因此 将像消息队列发送订单信息的操作，也写在Lua 脚本中

    ```Lua
     -- 1. 参数列表
     -- 优惠券Id
     local voucherId = ARGV[1]
     -- 用户Id
     local userId = ARGV[2]
     -- ⭐订单Id
     local orderId = ARGV[3]
     -- 2. 数据key
     -- 库存key
     local stockKey = "seckill:stock:" .. voucherId
     -- 订单key
     local orderKey = "seckill:order:" .. voucherId
     -- 3. 脚本业务
     -- 3.1 判断库存是否充足 get stockKey
     if(tonumber(redis.call('get', stockKey)) <= 0) then
         -- 库存不足返回1
         return 1
     end
     -- 3.2 判断用户是否下单 sismember orderKey userId
     if (tonumber(redis.call('sismember', orderKey, userId)) == 1) then
         -- 存在，即重复下单，返回2
         return 2
     end
     -- 3.4 扣库存 incrby stockKey -1
     redis.call('incrby', stockKey, -1)
     -- 3.5 下单（保存用户） sadd orderKey userId
     redis.call('sadd', orderKey, userId)
     -- ⭐3.5 （生产者）发送消息到队列中 XADD stream.orders * k1 v1 k2 v2
     redis.call('xadd','stream.orders','*','userId',userId,'voucherId',voucherId,'id',orderId)
     return 0
    ```
3. 最终完整的 VoucherOrderServiceImpl

    ```Java
     @Slf4j
     @Service
     public class VoucherOrderServiceImpl extends ServiceImpl<VoucherOrderMapper, VoucherOrder> implements IVoucherOrderService {
         @Resource
         private ISeckillVoucherService SeckillVoucherService;
         @Resource
         private RedisWorker redisWorker;
         @Resource
         private StringRedisTemplate stringRedisTemplate;
         private static final DefaultRedisScript<Long> SECKILL_SCRIPT;
         static {
             SECKILL_SCRIPT = new DefaultRedisScript<>();
             SECKILL_SCRIPT.setLocation(new ClassPathResource("seckill.lua"));
             SECKILL_SCRIPT.setResultType(Long.class);
         }
         //创建异步线程，也是消息队列中的🤏消费者角色
         private static final ExecutorService SECKILL_ORDER_EXECUTOR = Executors.newSingleThreadExecutor();
         //加载类时，异步线程就去阻塞队列中读取订单数据
         @PostConstruct
         private void init(){
             SECKILL_ORDER_EXECUTOR.submit(new VoucherOrderHandler());
         }
         //异步线程需要执行的Task
         private class VoucherOrderHandler implements Runnable{
             @Override
             public void run() {
                 while (true){
                     try {
                         //1. ⭐获取消息队列中的订单信息 XREADGROUP GROUP g1 c1 COUNT 1 BLOCK 2000 STREAM s1 >
                         List<MapRecord<String, Object, Object>> list = stringRedisTemplate.opsForStream().read(
                                 Consumer.from("g1", "c1"),
                                 StreamReadOptions.empty().count(1).block(Duration.ofSeconds(2)),
                                 StreamOffset.create("stream.orders", ReadOffset.lastConsumed())
                         );
                         //2. 判断订单消息是否为空
                         if (list == null || list.isEmpty()) {
                             //2.1 如果为null，说明当前消息队列中没有消息，继续下一次循环阻塞读取
                             continue;
                         }
                         //2.2 消息非空，将消息解析为优惠券对象
                         MapRecord<String, Object, Object> record = list.get(0);
                         Map<Object, Object> value = record.getValue();
                         VoucherOrder voucherOrder = BeanUtil.fillBeanWithMap(value, new VoucherOrder(), true);
                         //3 创建订单
                         createVoucherOrder(voucherOrder);
                         //3. 确认消息 XACK s1 g1 id
                         stringRedisTemplate.opsForStream().acknowledge("stream.orders","g1",record.getId());
                     } catch (Exception e) {
                         log.error("处理订单时异常",e);
                         //如果上面的代码出现异常，还需要处理pending-list中未确认的异常消息
                         handlePendingList();
                     }
                 }
             }
         }
         //⭐当消息处理异常时，需要处理 handlePendingList 中未确认的消息
         private void handlePendingList() {
             while (true){
                 try {
                     //1. 获取pending-list中的订单信息 XREADGROUP GROUP g1 c1 COUNT 1 BLOCK 2000 STREAM s1 0
                     List<MapRecord<String, Object, Object>> list = stringRedisTemplate.opsForStream().read(
                             Consumer.from("g1", "c1"),
                             StreamReadOptions.empty().count(1),
                             StreamOffset.create("stream.orders", ReadOffset.from("0"))
                     );
                     //2. 判断获取到的pending-list消息是否为空
                     if (list == null || list.isEmpty()) {
                         //2.1 如果为null，说明当前pending-list中没有消息，直接结束，回到VoucherOrderHandler的while逻辑，继续从消息队列中阻塞获取
                         break;
                     }
                     //2.2 消息非空，将消息解析为优惠券对象
                     MapRecord<String, Object, Object> record = list.get(0);
                     Map<Object, Object> value = record.getValue();
                     VoucherOrder voucherOrder = BeanUtil.fillBeanWithMap(value, new VoucherOrder(), true);
                     //3 创建订单
                     createVoucherOrder(voucherOrder);
                     //3. 确认消息 XACK s1 g1 id
                     stringRedisTemplate.opsForStream().acknowledge("stream.orders","g1",record.getId());
                 } catch (Exception e) {
                     log.error("处理订单时异常",e);
                     //即使在处理pending-list中的消息时出现异常，也无需再处理
                     //就让它无限循环，直到pending-list中的消息全部解析完，pending-list为空 break结束
                     //为了避免频率太高，可以做个休眠
                     try {
                         Thread.sleep(100);
                     } catch (InterruptedException interruptedException) {
                         interruptedException.printStackTrace();
                     }
                 }
             }
         }
         //操作数据库：扣减库存，保存订单
         private void createVoucherOrder(VoucherOrder voucherOrder) {
             //扣减库存
             boolean success = SeckillVoucherService.update()
                     .setSql("stock = stock - 1")
                     .eq("voucher_id", voucherOrder.getId())
                     .gt("stock",0)
                     .update();
             if (!success) {
                 log.error("库存不足！");
                 return;
             }
             //保存订单
             save(voucherOrder);
         }
         // ⭐业务入口，也是消息队列中的📦生产者角色
         @Override
         public Result seckillVoucher(Long voucherId) {
             Long userId = UserHolder.getUser().getId();
             long voucherOrderId = redisWorker.nextID("order");
             //1. 执行lua脚本
             Long result = stringRedisTemplate.execute(SECKILL_SCRIPT, Collections.emptyList(),
                     voucherId.toString(), userId.toString(),String.valueOf(voucherOrderId));
             //2. 判断结果是否为0
             assert result != null;
             int r = result.intValue();
             if (r != 0) {
                 //3. 不为0，代表没有购买资格
                 return Result.fail(r == 1 ? "库存不足" : "不能重复下单");
             }
             //3. 生产者向stream队列中发送消息已经在lua脚本中完成，如果返回的是0，说明消息已经发送个成功
             //4. 返回订单Id
             //⭐由于真正对数据库的操作是交给异步线程执行的，而异步线程在类加载时就会执行，故这里可以直接将生成的订单信息返回给用户
             return Result.ok(voucherOrderId);
         }
     }
    ```
