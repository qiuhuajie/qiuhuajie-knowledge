# 二、系统初始化
1. 修改静态IP

    ```Plain
     vi /etc/sysconfig/network-scripts/ifcfg-ens33
    ```
2. 修改主机名

    ```Plain
     vi /etc/hostname
    ```
3. 添加域名映射

    ```Plain
     vi /etc/hosts
    ```
4. 重启生效

    ```Plain
     systemctl restart network
     reboot
    ```
5. 保证三个节点之间可以使用主机名 Ping 通
6. 关闭防火墙

    ```Plain
     systemctl stop firewalld
     systemctl disable firewalld
     systemctl status firewalld
    ```
7. 关闭 SE Linux

    ```Plain
     # 永久关闭
     sed -i 's/enforcing/disabled/' /etc/selinux/config
     # 临时关闭
     setenforce 0
    ```
8. 关闭 swap

    ```Plain
     # 永久关闭
     sed -ri 's/.*swap.*/#&/' /etc/fstab         # 需要重启后才能生效
     # 临时关闭
     swapoff -a
     # 验证：
     [root@master ~]# free -m
                   total        used        free      shared  buff/cache   available

     Swap:             0           0           0
    ```
9. 将桥接的 IPv4 流量传递到 iptables 的链

    ```Plain

     net.bridge.bridge-nf-call-ip6tables = 1net.bridge.bridge-nf-call-iptables = 1
     EOF
     sysctl --system
    ```

# 三、部署 Etcd 集群
## 1. 生成自签证书
1. 安装证书生成工具
    > cfssl 是一个开源的证书管理工具， 使用 json 文件生成证书， 相比 openssl 更方便使用
    > 在 master 节点上操作
    ```Plain
     mkdir -p ~/TLS/{etcd,k8s}
     cd TLS
    ```

    下载

    ```Plain
     wget https://pkg.cfssl.org/R1.2/cfssl_linux-amd64   –no-check-certificate
     wget https://pkg.cfssl.org/R1.2/cfssljson_linux-amd64   –no-check-certificate
     wget https://pkg.cfssl.org/R1.2/cfssl-certinfo_linux-amd64   –no-check-certificate
     [root@master TLS]# ls
     cfssl-certinfo_linux-amd64  cfssljson_linux-amd64  cfssl_linux-amd64  etcd  k8s
    ```

    赋予执行权限，并加到系统环境变量中

    ```Plain
     chmod +x cfssl_linux-amd64 cfssljson_linux-amd64 cfssl-certinfo_linux-amd64
     mv cfssl_linux-amd64 /usr/local/bin/cfssl
     mv cfssljson_linux-amd64 /usr/local/bin/cfssljson
     mv cfssl-certinfo_linux-amd64 /usr/bin/cfssl-certinfo
     [root@master TLS]# ls
     etcd  k8s
    ```
2. 生成证书
    1. 进入工作目录

        ```Plain
         cd TLS/etcd
        ```
    2. 创建自签证书颁发机构（CA）

        ```Plain
         cat > ca-config.json << EOF
         {
           "signing": {
             "default": {
               "expiry": "87600h"
             },
             "profiles": {
               "www": {
                  "expiry": "87600h",
                  "usages": [
                     "signing",
                     "key encipherment",
                     "server auth",
                     "client auth"
                 ]
               }
             }
           }
         }
         EOF
        ```
        ```Plain
         cat > ca-csr.json << EOF
         {
             "CN": "etcd CA",
             "key": {
                 "algo": "rsa",
                 "size": 2048
             },
             "names": [
                 {
                     "C": "CN",
                     "L": "Beijing",
                     "ST": "Beijing"
                 }
             ]
         }
         EOF
        ```
    3. 使用 cfssl 生成证书

        ```Plain
         cfssl gencert -initca ca-csr.json | cfssljson -bare ca -
        ```
    4. 查看生成的证书

        ```Plain
         [root@master etcd]# ls *pem
         ca-key.pem  ca.pem
        ```
3. 使用自签 CA 签发 Etcd HTTPS 证书
    1. 创建证书申请文件（需要将 hosts 改成master 和 node 的IP）

        ```Plain
         cat > server-csr.json << EOF
         {
             "CN": "etcd",
             "hosts": [
             "192.168.10.181",
             "192.168.10.182"
             ],
             "key": {
                 "algo": "rsa",
                 "size": 2048
             },
             "names": [
                 {
                     "C": "CN",
                     "L": "BeiJing",
                     "ST": "BeiJing"
                 }
             ]
         }
         EOF
        ```
    2. 生成证书

        ```Plain
         cfssl gencert -ca=ca.pem -ca-key=ca-key.pem -config=ca-config.json -profile=www server-csr.json | cfssljson -bare server
        ```
4. 查看最终生成的证书

    ```Plain
     [root@master etcd]# ls *pem
     ca-key.pem  ca.pem  server-key.pem  server.pem
    ```
## 2. 部署 Etcd 集群
1. 下载 etcd
2. 创建工作目录（先将所有的东西在`~/etcd/`中准备好，再`mv`到`/opt/etcd/`），并解压下载的压缩包

    ```Plain
     [root@master ~]# mkdir etcd
     [root@master ~]# ls
     anaconda-ks.cfg  etcd  –p  TLS
     [root@master ~]# mkdir ./etcd/{bin,cfg,ssl}
     [root@master ~]# cd etcd/
     [root@master etcd]# ls
     bin  cfg  ssl
    ```
    ```Plain
     cd ~
     tar zxvf etcd-v3.4.9-linux-amd64.tar.gz
     [root@master ~]# ls
     anaconda-ks.cfg  etcd  etcd-v3.4.9-linux-amd64  etcd-v3.4.9-linux-amd64.tar.gz  –p  TLS
    ```
3. 给 `~/etcd/bin`下放东西

    ```Plain
     [root@master ~]# mv etcd-v3.4.9-linux-amd64/{etcd,etcdctl} ./etcd/bin/
     [root@master ~]# cd etcd/bin/
     [root@master bin]# ls
     etcd  etcdctl
    ```
4. 给 `~/etcd/cfg`下放东西

    注意修改为本机 IP，`ETCD_INITIAL_CLUSTER`中为etcd集群中的所有主机IP

    ```Plain
     cat > ~/etcd/cfg/etcd.conf << EOF
     #[Member]
     ETCD_NAME="etcd-1"
     ETCD_DATA_DIR="/var/lib/etcd/default.etcd"
     ETCD_LISTEN_PEER_URLS="https://192.168.10.181:2380"
     ETCD_LISTEN_CLIENT_URLS="https://192.168.10.181:2379"
     #[Clustering]
     ETCD_INITIAL_ADVERTISE_PEER_URLS="https://192.168.10.181:2380"
     ETCD_ADVERTISE_CLIENT_URLS="https://192.168.10.181:2379"
     ETCD_INITIAL_CLUSTER="etcd-1=https://192.168.10.181:2380,etcd-2=https://192.168.10.182:2380"
     ETCD_INITIAL_CLUSTER_TOKEN="etcd-cluster"
     ETCD_INITIAL_CLUSTER_STATE="new"
     EOF
    ```
5. 给 `~/etcd/ssl`下放东西（将生成的所有证书放入）

    ```Plain
     [root@master etcd]# cp ~/TLS/etcd/ca*pem ~/TLS/etcd/server*pem ~/etcd/ssl/
    ```
6. systemd 管理 etcd：添加 `/usr/lib/systemd/system/etcd.service`

    ```Plain
     cat > /usr/lib/systemd/system/etcd.service << EOF
     [Unit]
     Description=Etcd Server
     After=network.target
     After=network-online.target
     Wants=network-online.target
     [Service]
     Type=notify
     EnvironmentFile=/opt/etcd/cfg/etcd.conf
     ExecStart=/opt/etcd/bin/etcd \
     --cert-file=/opt/etcd/ssl/server.pem \
     --key-file=/opt/etcd/ssl/server-key.pem \
     --peer-cert-file=/opt/etcd/ssl/server.pem \
     --peer-key-file=/opt/etcd/ssl/server-key.pem \
     --trusted-ca-file=/opt/etcd/ssl/ca.pem \
     --peer-trusted-ca-file=/opt/etcd/ssl/ca.pem \
     --logger=zap
     Restart=on-failure
     LimitNOFILE=65536
     [Install]
     WantedBy=multi-user.target
     EOF
    ```
7. 上面 etcd.service 中写的 service 启动是在/opt/etcd，故将`~/etcd/` `mv`到`/opt/etcd/`

    ```Plain
     mv ~/etcd /opt/
    ```
    ```Plain
     # yum install tree
     [root@master etcd]# tree /opt/etcd
     /opt/etcd
     ├── bin
     │   ├── etcd
     │   └── etcdctl
     ├── cfg
     │   └── etcd.conf
     └── ssl
         ├── ca-key.pem
         ├── ca.pem
         ├── server-key.pem
         └── server.pem
    ```
8. 再将 master 节点上的`/opt/etcd/`，分发给所有 node 节点

    ```Plain
     scp -r /opt/etcd/ root@192.168.10.182:/opt/
    ```
9. 将 master 节点上的`/usr/lib/systemd/system/etcd.service`，分发给所有 node 节点

    ```Plain
     scp /usr/lib/systemd/system/etcd.service root@192.168.10.182:/usr/lib/systemd/system/
    ```
10. 在node 节点上将 接受到的 `etcd/cfg/etcd.conf` 中的IP改成node本机IP

    注意：`ETCD_NAME`也需要修改

    ```Plain
     vi /opt/etcd/cfg/etcd.conf
     #[Member]
     ETCD_NAME="etcd-2"
     ETCD_DATA_DIR="/var/lib/etcd/default.etcd"
     ETCD_LISTEN_PEER_URLS="https://192.168.10.182:2380"
     ETCD_LISTEN_CLIENT_URLS="https://192.168.10.182:2379"
     #[Clustering]
     ETCD_INITIAL_ADVERTISE_PEER_URLS="https://192.168.10.182:2380"
     ETCD_ADVERTISE_CLIENT_URLS="https://192.168.10.182:2379"
     ETCD_INITIAL_CLUSTER="etcd-1=https://192.168.10.181:2380,etcd-2=https://192.168.10.182:2380"
     ETCD_INITIAL_CLUSTER_TOKEN="etcd-cluster"
     ETCD_INITIAL_CLUSTER_STATE="new"
    ```
11. 在 master上启动 etcd

    ```Plain
     systemctl daemon-reload
     systemctl enable etcd.service
     systemctl start etcd.service
     systemctl status etcd.service
    ```

    ```Plain
     systemctl status etcd.service
     ● etcd.service - Etcd Server
    ```

## 3. 部署 apiserver

    ```Plain

     --enable-bootstrap-token-auth=true \\

     --service-node-port-range=30000-32767 \\
     --kubelet-client-certificate=/opt/kubernetes/ssl/server.pem \\
     --kubelet-client-key=/opt/kubernetes/ssl/server-key.pem \\
     --tls-cert-file=/opt/kubernetes/ssl/server.pem  \\
     --tls-private-key-file=/opt/kubernetes/ssl/server-key.pem \\
     --client-ca-file=/opt/kubernetes/ssl/ca.pem \\
     --service-account-key-file=/opt/kubernetes/ssl/ca-key.pem \\
     --etcd-cafile=/opt/etcd/ssl/ca.pem \\
     --etcd-certfile=/opt/etcd/ssl/server.pem \\
     --etcd-keyfile=/opt/etcd/ssl/server-key.pem \\
     --audit-log-maxage=30 \\
     --audit-log-maxbackup=3 \\
     --audit-log-maxsize=100 \\
     --audit-log-path=/opt/kubernetes/logs/k8s-audit.log"
     EOF
    ```
    > -- logtostderr： 启用日志
    > -- v： 日志等级
    > -- log-dir： 日志目录
    > -- etcd-servers： etcd 集群地址
    > -- bind-address： 监听地址
    > -- secure-port： https 安全端口
    > -- advertise-address： 集群通告地址
    > -- allow-privileged： 启用授权
    > -- service-cluster-ip-range： Service 虚拟 IP 地址段
    > -- enable-admission-plugins： 准入控制模块
    > -- authorization-mode： 认证授权， 启用 RBAC 授权和节点自管理
    > -- enable-bootstrap-token-auth： 启用 TLS bootstrap 机制
    > -- token-auth-file： bootstrap token 文件
    > -- service-node-port-range： Service nodeport 类型默认分配端口范围
    > -- kubelet-client-xxx： apiserver 访问 kubelet 客户端证书
    > -- tls-xxx-file： apiserver https 证书
    > -- etcd-xxxfile： 连接 Etcd 集群证书
    > -- audit-log-xxx： 审计日志
