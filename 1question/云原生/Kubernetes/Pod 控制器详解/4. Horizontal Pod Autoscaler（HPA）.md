> [!important] **手动调整 Pod 规模 ➡️ 自动调整 Pod 规模**
1. 上面 **Deploy 控制器** 可以实现通过手工执行**`kubectl scale`**命令实现Pod扩容或缩容
1. 但是这显然不符合 Kubernetes的定位目标：自动化、智能化
1. ⭐**期望**==**可以实现通过监测Pod的使用情况，实现**== ==**pod 数量的自动调整**==**，于是就产生了Horizontal Pod Autoscaler（HPA）这种控制器**
    
    1. **HPA可以获取每个Pod利用率，然后和HPA中定义的指标进行对比**
    
    1. **同时计算出需要伸缩的具体值，最后实现Pod的数量的调整**
    
1. HPA 实现原理
    
    1. HPA与之前的Deployment一样，也属于一种Kubernetes资源对象
    
    1. 通过**追踪分析RC控制的所有目标Pod的负载变化情况**，来**确定是否需要针对性地调整目标Pod的副本数**

1. 示例：
    
    1. **安装 metrics-server**
        
        1. 安装 git
            
            ```Plain
             [root@master ~]# yum install git -y
            ```
            
        
        1. clone 源码
            
            ```Plain
             [root@master ~]# git clone -b v0.3.6 https://github.com/kubernetes-incubator/metrics-server
            ```
            
        
        1. 修改 yaml 配置文件
            
            ```Plain
             [root@master ~]# cd /root/metrics-server/deploy/1.8+/
             [root@master 1.8+]# pwd
             /root/metrics-server/deploy/1.8+
             [root@master 1.8+]# vim metrics-server-deployment.yaml
            ```
            
            添加配置项
            
            ```Plain
             ---
             apiVersion: v1
             kind: ServiceAccount
             metadata:
               name: metrics-server
               namespace: kube-system
             ---
             apiVersion: apps/v1
             kind: Deployment
             metadata:
               name: metrics-server
               namespace: kube-system
               labels:
                 k8s-app: metrics-server
             spec:
               selector:
                 matchLabels:
                   k8s-app: metrics-server
               template:
                 metadata:
                   name: metrics-server
                   labels:
                     k8s-app: metrics-server
                 spec:
                   hostNetwork: true     # 添加
                   serviceAccountName: metrics-server
                   volumes:
                   - name: tmp-dir
                     emptyDir: {}
                   containers:
                   - name: metrics-server
                     image: registry.cn-hangzhou.aliyuncs.com/google_containers/metrics-server-amd64:v0.3.6      # 添加
                     imagePullPolicy: Always
                     args:   # 添加
                     - --kubelet-insecure-tls
                     - --kubelet-preferred-address-types=InternalIP,Hostname,InternalDNS,ExternalDNS,ExternalIP
                     volumeMounts:
                     - name: tmp-dir
                       mountPath: /tmp
            ```
            
        
        1. 安装 metrics-server
            
            ```Plain
             [root@master 1.8+]# kubectl apply -f ./
            ```
            
        
        1. 查看 Pod 运行情况
            


        1. metrics-server 安装成功
        
    
    1. **准备deployment和servie**
        
        1. 创建pc-hpa-pod.yaml文件
            
            ```Plain
             apiVersion: apps/v1
             kind: Deployment
             metadata:
               name: nginx
               namespace: dev
             spec:
               strategy: # 策略
                 type: RollingUpdate # 滚动更新策略
               replicas: 1
               selector:
                 matchLabels:
                   app: nginx-pod
               template:
                 metadata:
                   labels:
                     app: nginx-pod
                 spec:
                   containers:
                   - name: nginx
                     image: nginx:1.17.1
                     resources: # 资源配额
                       limits:  # 限制资源（上限）
                         cpu: "1" # CPU限制，单位是core数
                       requests: # 请求资源（下限）
                         cpu: "100m"  # CPU限制，单位是core数
            ```
            
        
        1. 创建deployment
            
            ```Plain
             [root@master ~]# kubectl create -f pc-hpa-pod.yaml -n dev
             deployment.apps/nginx created
            ```
            
        
        1. 创建service（让 Pod 对外暴露端口提供服务）
            
            ```Plain
             [root@master ~]# kubectl expose deployment nginx --type=NodePort --port=80 -n dev
             service/nginx exposed
            ```
            
        
        1. 查看 deploy、service、pod
            
            ```Plain
             [root@master ~]# kubectl get deployment,pod,svc -n dev
             NAME                    READY   UP-TO-DATE   AVAILABLE   AGE
             deployment.apps/nginx   1/1     1            1           4m1s
             
             NAME                        READY   STATUS    RESTARTS   AGE
             pod/nginx-cd5f9b74c-cqkxs   1/1     Running   0          4m1s
             
             NAME            TYPE       CLUSTER-IP      EXTERNAL-IP   PORT(S)        AGE
             service/nginx   NodePort   10.96.201.184   <none>        80:32708/TCP   43s
            ```
            
        
        1. 确保可以使用 PodIP:30634 访问到 nginx
        
    
    1. **部署HPA**
        
        1. 创建pc-hpa.yaml文件
            
            ```Plain
             apiVersion: autoscaling/v1
             kind: HorizontalPodAutoscaler
             metadata:
               name: pc-hpa
               namespace: dev
             spec:
               minReplicas: 1  \#最小pod数量
               maxReplicas: 10 \#最大pod数量
               targetCPUUtilizationPercentage: 3 # CPU使用率指标
               scaleTargetRef:   # 指定要控制的nginx信息
                 apiVersion:  apps/v1
                 kind: Deployment
                 name: nginx
            ```
            
        
        1. 创建hpa
            
            ```Plain
             [root@master ~]# kubectl create -f pc-hpa.yaml
             horizontalpodautoscaler.autoscaling/pc-hpa created
            ```
            
        
        1. 查看hpa
            
            ```Plain
             [root@master ~]# kubectl get hpa -n dev
             NAME     REFERENCE          TARGETS        MINPODS   MAXPODS   REPLICAS   AGE
             pc-hpa   Deployment/nginx   <unknown>/3%   1         10        0          4s
             
             # 等待一会
             [root@master ~]# kubectl get hpa -n dev

             pc-hpa   Deployment/nginx   0%/3%     1         10        1          19s
             # TARGETS 当前负载百分比/额定负载百分比
            ```
