# 一、Docker 安装
1. 下载安装包：
    > ℹ️ Index of linux/centos/7/x86_64/stable/Packages/
    > containerd.
    > [https://download.docker.com/linux/centos/7/x86_64/stable/Packages/](https://download.docker.com/linux/centos/7/x86_64/stable/Packages/)
    ```Bash
     \#下载这两个包
     docker-ce-17.03.0.ce-1.el7.centos.x86_64.rpm
     docker-ce-selinux-17.03.0.ce-1.el7.centos.noarch.rpm
    ```
2. 关闭防火墙：

    ```Bash
     [root@docker001 qhj]# systemctl stop firewalld
     [root@docker001 qhj]# systemctl disable firewalld
    ```
3. 关闭SELinux

    ```Bash
     [root@docker001 qhj]# getenforce
     Enforcing
     [root@docker001 qhj]# vim /etc/selinux/config
     \#重启生效
     [root@docker001 qhj]# reboot
     [root@docker001 ~]# getenforce
     Disabled
    ```
4. 安装iptables管理防火墙

    ```Bash
     [root@docker001 ~]# yum -y install iptables-services
     [root@docker001 ~]# systemctl start iptables
     \#设为开机自启
     [root@docker001 ~]# systemctl enable iptables
     Created symlink from /etc/systemd/system/basic.target.wants/iptables.service to /usr/lib/systemd/system/iptables.service.
     \#清空iptables中的规则，并保存
     [root@docker001 ~]# iptables -F
     [root@docker001 ~]# service iptables save
     iptables: Saving firewall rules to /etc/sysconfig/iptables:[  确定  ]
    ```
5. 更新操作系统软件包

    ```Plain
     [root@docker001 ~]# yum -y update
     \#重启系统
     [root@docker001 ~]# reboot
    ```
6. 将下载好的docker安装包移动到创建的docker目录下

    ```Plain
     \#拖拽两个文件即可
     [root@docker001 ~]# rz -E
     rz waiting to receive.
     [root@docker001 ~]# mkdir docker
     [root@docker001 ~]# ls
     anaconda-ks.cfg  docker  initial-setup-ks.cfg  公共  模板  视频  图片  文档  下载  音乐  桌面
     [root@docker001 ~]# cd docker/
     [root@docker001 docker]# ls
     docker-ce-17.03.0.ce-1.el7.centos.x86_64.rpm  docker-ce-selinux-17.03.0.ce-1.el7.centos.noarch.rpm
    ```
7. 执行docker安装

    ```Plain
     [root@docker001 docker]# yum -y install *
    ```
8. 启动docker

    ```Plain
     [root@docker001 docker]# systemctl start docker
     \#设为开机自启
     [root@docker001 docker]# systemctl enable docker
     Created symlink from /etc/systemd/system/multi-user.target.wants/docker.service to /usr/lib/systemd/system/docker.service.
    ```
9. 查看docker状态

    ```Plain
     [root@docker001 docker]# systemctl status docker
     ● docker.service - Docker Application Container Engine
        Loaded: loaded (/usr/lib/systemd/system/docker.service; enabled; vendor preset: disabled)
    ```

# 二、配置 阿里云加速
1. 在`docker.service`中配置阿里源（阿里源在本地和远程官方源中间提供加速）

    ```Bash
    cp /lib/systemd/system/docker.service /etc/systemd/system/
    chmod a+x /etc/systemd/system/docker.service
    vim /etc/systemd/system/docker.service
    ```
    ```Bash
     cat /etc/systemd/system/docker.service
     [Unit]
     Description=Docker Application Container Engine
     Documentation=https://docs.docker.com
     After=network.target firewalld.service
     [Service]
     Type=notify
     \#在此处添加 --registry-mirror=https://kf[63jaj.mirror.aliyuncs.com
     ExecStart=/usr/bin/dockerd --registry-mirror=https://kf[63jaj.mirror.aliyuncs.com
     ExecReload=/bin/kill -s HUP $MAINPID
     ...
    ```
2. 重启docker

    ```Bash
     \#重载管理脚本
    systemctl daemon-reload
     \#重启docker
    systemctl restart :wqdocker
    ```
3. 验证是否配置完成

    ```Bash
    ps -ef | grep docker
     avahi       719      1  0 20:02 ?        00:00:00 avahi-daemon: running [docker001.local]
     root       2444      1  0 20:26 ?        00:00:00 /usr/bin/dockerd --registry-mirror=https://kf[63jaj.mirror.aliyuncs.com
     \#配置完成
    ```
4. 或者使用 `docker info` 验证

    ```Plain
     docker info
     ...
     Registry Mirrors:
      https://b9pmyelo.mirror.aliyuncs.com/
     ...
    ```
